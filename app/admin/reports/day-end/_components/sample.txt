"use client"

import { useState } from 'react';
import * as XLSX from 'xlsx';
import { PDFDownloadLink } from '@react-pdf/renderer';

import { Button } from '@/components/ui/button';
import { fetchGmtDefects } from "@/actions/admin/fetch-gmt-defetcs";
import { calculateDhuAndAcv } from "@/actions/qc/gmt/calculate-dhu-acv";
import { fetchObbSheetDetails } from '@/actions/admin/fetch-obb-sheet-details';
import SelectScanningPointAndDate from "@/components/forms/select-scanning-point-and-date";
import HourlyQuantityReportTemplate from '@/components/templates/report/hourly-quantity-report-template';
import HourlyQuantityReportViewer from '@/components/templates/report/viewer/hourly-quantity-report-viewer';

interface DayEndReportProps {
    scanningPoints: {
        id: string;
        name: string;
        pointNo: number;
    }[] | null;
    userName: string;
}

type GmtQcReportDetailsType = {
    label: string;
    value: string;
};

type GmtQcHourlyQuantityType = {
    hourGroup: string;
    inspectQty: number;
    passQty: number;
    reworkQty: number;
    rejectQty: number;
    DHU: number;
    ACV: number;
};

const DayEndReport = ({
    scanningPoints,
    userName
}: DayEndReportProps) => {
    const [pdfLink, setPdfLink] = useState<JSX.Element | null>(null);
    const [excellSheetDetails, setExcellSheetDetails] = useState({ sheetName: "", fileName: "" });
    const [obbSheet, setObbSheet] = useState<ObbSheetDetailsType>({id:"", color:null, style:"", unitName:"", lineName:""});
    const [gmtQcHourlyQuantity, setGmtQcHourlyQuantity] = useState<GmtQcHourlyQuantityType[]>([]);
    const [gmtQcReportDetails, setGmtQcReportDetails] = useState<GmtQcReportDetailsType[]>([]);
    const [show, setShow] = useState(false);

    const generateExcelSheet = () => {
        if (gmtQcHourlyQuantity.length > 0 && excellSheetDetails.fileName && excellSheetDetails.sheetName) {
            const worksheet = XLSX.utils.json_to_sheet(gmtQcHourlyQuantity);
            const workbook = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(workbook, worksheet, excellSheetDetails.sheetName);
            XLSX.writeFile(workbook, excellSheetDetails.fileName);
        }
    }

    const generatePdfReport = (hourlyQuantity: any[], totalDHU: number, date: string, scanningPointId: string) => {
        const reportDetails = [
            { label: "Scanning Point", value: scanningPoints?.find(p => p.id === scanningPointId)?.name || '' },
            { label: "Date", value: date },
            { label: "Generated By", value: userName },
            { label: "Style", value: obbSheet.style },
            { label: "Color", value: obbSheet.color || "" },
            { label: "Unit", value: obbSheet.unitName },
            { label: "Line", value: obbSheet.lineName },
            // { label: "Name of QI", value: "QI Name" },
            { label: "Total DHU", value: totalDHU.toFixed(2) },
        ];
        setGmtQcReportDetails(reportDetails);
        console.log("OBB State", obbSheet);
        
        return (
            <PDFDownloadLink
                document={
                    <HourlyQuantityReportTemplate
                        details={reportDetails}
                        data={hourlyQuantity}
                    />
                }
                fileName="qc-report.pdf"
            >
                {({ loading }) => (loading ? "Generating PDF..." : "Download PDF Report")}
            </PDFDownloadLink>
        ); 
    }

    const generateReportForGmtQc = async (scanningPointId: string, date: string) => {
        const res = await fetchGmtDefects(scanningPointId, date);
        const { totalDHU, hourlyQuantity } = calculateDhuAndAcv(res, 16);
        console.log("totalDHU:", totalDHU.toFixed(2));
        setGmtQcHourlyQuantity(hourlyQuantity);
        setExcellSheetDetails({ sheetName: "Hourly QC Report", fileName: "qc-report.pdf" });
        
        const pdfElement = generatePdfReport(hourlyQuantity, totalDHU, date, scanningPointId);
        setPdfLink(pdfElement);
    }

    const handleGenerateReport = async (data: { obbSheetId: string; scanningPointId: string; pointNo: number; date: Date }) => {
        data.date.setDate(data.date.getDate() + 1);
        const formattedDate = data.date.toISOString().split('T')[0];

        const obbRes = await fetchObbSheetDetails(data.obbSheetId);
        if (obbRes) {
            setObbSheet({ id: obbRes.id, style: obbRes.style, color: obbRes.color, unitName: obbRes.unitName, lineName: obbRes.lineName });
            console.log("OBB Res", obbRes);
        }
        
        if (data.pointNo === 5 || data.pointNo === 6) {
            await generateReportForGmtQc(data.scanningPointId, formattedDate);
        }
    }

    return (
        <div className="mx-auto max-w-7xl">
            <SelectScanningPointAndDate 
                scanningPoints={scanningPoints}
                handleSubmit={handleGenerateReport}
            />
            <button onClick={() => setShow(true)}>show</button>
            {pdfLink && (
                <Button className="mt-4">
                    {pdfLink}
                </Button>
            )}
            {(gmtQcReportDetails.length > 0 && gmtQcHourlyQuantity.length > 0) && 
                <div className='mt-8 p-8 bg-slate-100 rounded-lg border'>
                    <div className=''>
                        <HourlyQuantityReportViewer 
                            details={gmtQcReportDetails}
                            data={gmtQcHourlyQuantity}
                        />
                    </div>
                </div>
            }
        </div>
    )
}

export default DayEndReport
